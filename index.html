void refill() {
                  auto now = std::chrono::steady_clock::now();
                  double seconds = std::chrono::duration(now - lastRefill).count();
                  int newTokens = static_cast(seconds * refillRate);
                  
                  if (newTokens > 0) {
                      tokens = std::min(capacity, tokens + newTokens);
                      lastRefill = now;
                  }
              }
          
          public:
              RateLimiter(int cap, int rate) : capacity(cap), tokens(cap), refillRate(rate) {
                  lastRefill = std::chrono::steady_clock::now();
              }
          
              bool allowRequest() {
                  refill();
                  if (tokens > 0) {
                      tokens--;
                      return true;
                  }
                  return false;
              }
          };
